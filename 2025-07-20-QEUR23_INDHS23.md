    # 累積ポイントを初期化
    accum_money = 0
    accum_relation = 0
    accum_health = 0
    accum_business = 0
    
    # DataFrame用のデータを準備
    df_data = []
    for p in periods:
        period_str = f"{p['year']}年{p['month']}月"
        points = p["points"]
        
        # 累積ポイントを更新
        accum_money += points.get("money", 0) + dic_adjust["money(value)"]
        accum_relation += points.get("relation", 0) + dic_adjust["relation(value)"]
        accum_health += points.get("health", 0) + dic_adjust["health(value)"]
        accum_business += points.get("business", 0) + dic_adjust["business(value)"]
        
        df_data.append({
            "期間": period_str,
            "イベント": p["event"],
            "影響": p["impact"],
            "説明": p["description"],
            "money(value)": points.get("money", 0),
            "money(Accumulation)": accum_money,
            "relation(value)": points.get("relation", 0),
            "relation(Accumulation)": accum_relation,
            "health(value)": points.get("health", 0),
            "health(Accumulation)": accum_health,
            "business(value)": points.get("business", 0),
            "business(Accumulation)": accum_business
        })
    
    # DataFrameを作成
    df = pd.DataFrame(df_data)
    
    if not df.empty:
        # 折れ線グラフを生成
        plot_cumulative_points(df)
    else:
        print("データがないため、グラフは生成されませんでした。")
    
    # Excelファイルに出力
    with pd.ExcelWriter(output_path, engine='xlsxwriter') as writer:
        df.to_excel(writer, sheet_name='Transit Analysis', index=False)

            # 画像を挿入する位置を計算

            worksheet = writer.sheets['Transit Analysis']
            
            # 画像を挿入
            worksheet.insert_image(f'A{img_row}', plot_path, {
                'x_scale': 0.8,
                'y_scale': 0.8
            })
    
    return df
